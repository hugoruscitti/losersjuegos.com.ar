Bueno, si te refieres a la animación, esta clase te puede ser útil:

//Aquí la clase
class Aniframe{
    int frame;
    int inicio;
    float delay;
    int ultima_act;
    int * lista_anima;
    int largo;
    public:
    Aniframe(int * l){
        frame = 0;
        inicio = SDL_GetTicks();
        delay = 1000/15;
        ultima_act = 0; //ultima
        lista_anima = l;
        /*
        int *IntPtr = &lista_anima[0];
        while( *IntPtr >= 0 ){
             IntPtr++;
             largo++;
             };*/
        largo = 0;
        while(lista_anima[largo]>=0){
            largo++;
            };
        };
    ~Aniframe(){};
    int Frame_es(bool Reset = 0){
        if (Reset){
            this->frame = 0;
            };
        if ((SDL_GetTicks() - this->ultima_act) > this->delay){
            this->frame ++;
            if (this->frame >= this->largo){
                this->frame = 0;
                }
            this->ultima_act = SDL_GetTicks();
            };
        return lista_anima[this->frame];
        };
    };

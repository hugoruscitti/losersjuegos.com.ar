   static unsigned char n=0, r=0;
   register unsigned char nn=0, nr;
   if(n == 0){

      n= Byte_Aleatorio_Recogida();
   }
   if(r == 0){

      r= Byte_Aleatorio_Recogida();
   }

   for(nr=0; nr < r; nr++){

      nn= Byte_Aleatorio_Recogida();

   }
   r= (r + nn) % 255;
   r= (r * r) % n;

   return r;
